        \looseitems
            \item Central hadronic calorimeter of the ATLAS detector covering $|\eta| < 1.7$
            \item Identification and energy measurements of hadrons, jets, tau leptons decaying hadronically and missing transverse energy, 
            \\resolution of $\sigma_\mathrm{E} / E = 56.4\%/\sqrt{E\mathrm{(GeV)}}\oplus5.5\%$

            \item 4 partitions: 2 halves of central long barrel (LB) \\and 2 extended barrels (EB)
            \item Modules: plastic scintillating tiles as active medium and steel plates as absorber

            \item Azimuthal direction: 64 modules in each barrel
            \item Typically, 2 PMTs (channels) for a given readout cell (total of 9852 channels, 5182 cells)
            \item Signals divided into 2 branches: low gain (LG) and high gain (HG) and amplified with a relative ratio of 1:64

\block{Calibration Systems}{
\vspace{-0.25cm}
    \begin{center}
            \includegraphics[width=1.\linewidth]{plots/plot5.pdf}
        \end{center}
        Various calibration systems to calibrate each step of the TileCal cell energy reconstruction:
        \vspace{0.2cm}
        
        \looseitems
            \item \textbf{\textcolor{cesium}{Cesium System}} - performed on a monthly basis
            \begin{itemize}
                \item $^{137}$Cs capsule hydraulically driven through all tiles
                \item Calibrates variations of the entire readout chain (optical components and PMTs)
            \end{itemize}
            
            \item \textbf{\textcolor{laser}{Laser System}} - updated every 2 weeks
            \begin{itemize}
                \item Short laser pulses simultaneously sent to all PMTs
                \item Calibrates variations due to electronics and PMTs
            \end{itemize}
            
            \item \textbf{\textcolor{cis}{Charge Injection System (CIS)}} - updated monthly
            \begin{itemize}
                \item Known charge injected into readout electronics, calibrates the response of ADCs
            \end{itemize}
            
            \item \textbf{\textcolor{green}{Minimum Bias system}} - updated yearly
            \begin{itemize}
                \item Uses minimum-bias $pp$ collisions for the cells most exposed to radiation
            \end{itemize}
        \end{itemize}
        
        \vspace{0.3cm}
        $\textcolor{turquoise}{C_{\text{pC}\rightarrow\text{GeV}}}$ determined by measuring the response to electrons in test beam campaigns $\rightarrow$ EM scale with a nominal value of 1.05 pC/GeV
        
        \vspace{0.3cm}
        \begin{center}
        \Large{
         $E \text{ [GeV]} = \frac{A\text{[ADC]}}{\textcolor{cis}{C_{\text{ADC}\rightarrow\text{pC}}} \times \textcolor{turquoise}{C_{\text{pC}\rightarrow\text{GeV}}} \times \textcolor{cesium}{C_{\text{Cs}}} \times \textcolor{green}{C_{\text{MB}}} \times \textcolor{laser}{C_{\text{Las}}}}$
         }
        \end{center}
    }    
    
    \column{0.333333}
    
    \block{Cesium System}{
        \looseitems
            \item Hydraulic system moves $^{137}$Cs capsule (radioactive $\gamma$-source with $E_{\gamma} = 0.662$ MeV) through the calorimeter system measuring the response of every single tile
            \item Signals readout by a dedicated integrator system
            \item For typical cells the precision $\sim 0.3\%$
            \item Down-drifts in the response variation: degradation in the fibers, scintillating tiles, and PMTs
            \item Highest degradation for A layer cells as they are closest to the beam pipe
            \item PMTs recovery in the periods without $pp$ collisions
        \end{itemize}
        
        \begin{center}
            \includegraphics[width=0.49\linewidth]{plots/plot6.png}
            \includegraphics[width=0.49\linewidth]{plots/plot6a.png}
        \end{center}
    }
    
    \block{Laser System}{
        \looseitems
            \item Single laser source: short 10 ns light pulses with \\$\lambda=$ 532 nm simultaneously distributed to all PMTs
            \item Precision of the laser calibration at the level of $\sim0.5\%$
            \item Better stability in time observed for the demonstrator module
        \end{itemize}
        
        \begin{center}
            \includegraphics[width=0.49\linewidth]{plots/plot8.pdf}
            \includegraphics[width=0.49\linewidth]{plots/plot8a.pdf}
        \end{center}
    }
    
    \block{Charge Injection System}{
            \looseitems
                \item Injects a well-defined charge signal to the readout electronics covering the full dynamic range
                \item Determines $\textcolor{cis}{C_{\text{ADC}\rightarrow\text{pC}}}$ conversion constant from sample amplitude (ADC counts) to charge units (pC)
                \item Precision of the CIS calibration at the level of $\sim0.7\%$
                \item Very stable over time
            \end{itemize}
            
            \begin{center}
                \includegraphics[width=0.49\linewidth]{plots/plot11.png}
                \includegraphics[width=0.49\linewidth]{plots/plot12.png}
            \end{center}
            }
            
    
    \block{Minimum Bias System}{
            \looseitems
                \item Inelastic $pp$ collisions with low momentum transfer
                \item Minimum bias events produce a PMT current proportional to the LHC luminosity
                \item Signal readout by the same integrator system as used by Cesium calibration
                \item Degradation studies of cells in the gap region (E cells) inaccessible to the cesium system
                \item Precision of the calibration at the level of $\sim1.2\%$
            \end{itemize}

\block{Isolated Muons - Energy Response}{
            \looseitems
                \item Isolated muons produced in $W \rightarrow \mu\nu$ decays from $pp$ collisions used for calorimeter performance tests
               % \item Uniformity and stability of the cells' response and the EM scale setting checks
               % \item Energy loss per unit path close to the minimum ionizing level $\rightarrow$ muon is an ideal particle to scan the entire TileCal volume
                \item Energy Response: ratio of data and MC for the truncated mean of the $\Delta E/\Delta x$ (muon energy loss per unit distance) calculated without considering 1\% of events in the upper tail
                \vspace{-0.4cm}
                \begin{equation}
                  R = \frac{\langle\Delta E/\Delta x\rangle_{F=1\%}^{\text{Data}}}{\langle\Delta E/\Delta x\rangle_{F=1\%}^{\text{MC}}}\nonumber
                \end{equation}
                
                \item Response uniformity in azimuthal direction measured by fitting the double ratio distribution to a Gaussian likelihood function:
                \begin{equation}
                  \mathcal{L} = \prod_{m=1}^{64} \frac{1}{\sqrt{2\pi\sqrt{\sigma_m^2 + s^2}}} \exp\left[-\frac{1}{2}\frac{(R_m - \mu)^2}{\sigma_m^2 + s^2}\right]\nonumber
                \end{equation}
                
                $\mu$ - average energy response,\\
                $s$ - non-uniformity estimator
                \item Good uniformity across azimuthal modules (on average $\sim 2.6\%$ for 2022 data and $\sim 2.2\%$ for 2023 data)
            \end{itemize}

    \block{Time Performance}{
            \looseitems
                \item Using reconstructed jets with $p_\mathrm{T}>$ 20 GeV
                \item Mean cell time decreases with cell energy due to less neutrons and slow hadronic components of shower 
                \item Transition between high and low gain at $\sim 22$ GeV
                \item Cell time resolution: $\sigma$ of a Gaussian fit of a timing distribution in a given energy slice  
                \item Worse resolution for the extended barrel due to larger cell size and for outermost D-layer
                \item Resolution better than 1 ns for cell energy $>$ 4 GeV
            \end{itemize}
