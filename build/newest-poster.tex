\block{The ATLAS Tile Calorimeter \& the HL-LHC Upgrade}{
    \looseitems
        \item TileCal is the central hadronic sampling calorimeter of ATLAS,
        covering $|\eta|<1.7$ with steel absorbers and scintillating plastic tiles.
        It is essential for hadronic jet reconstruction, missing transverse
        energy ($E_{\rm T}^{\rm miss}$), and $\tau$-lepton identification.
        \item The High-Luminosity LHC phase (HL-LHC, $\sqrt{s}=14\,\text{TeV}$) targets
        $3000\,\text{fb}^{-1}$ at peak luminosity $7.5\times10^{34}\,\text{cm}^{-2}\text{s}^{-1}$
        and $\langle\mu\rangle\approx200$ simultaneous $pp$ interactions per crossing.
        \item To handle the increased data rate, the entire on-detector electronics chain
        is replaced: $40\,\text{MHz}$ fully continuous, un-triggered digitization of all
        $\sim$10000 channels replaces the current Level-1 triggered scheme.
        \item Every component must be certified for a decade of operation without
        maintenance access against \textbf{\textcolor{tid}{TID}},
        \textbf{\textcolor{niel}{NIEL}}, and \textbf{\textcolor{see}{SEE/\textcolor{sel}{SEL}}}.
    \end{itemize}
    \vspace{0.12cm}
    \begin{center}
        \includegraphics[width=1.0\linewidth]{assets/fig_minidrawer_blockdiagram.png}
    \end{center}
    \vspace{-0.25cm}
    \captiontext{Phase-II Minidrawer (left) and on-/off-detector readout chain (right).
    Each module hosts 4 Minidrawers, each servicing up to 12 PMTs at $40\,\text{MHz}$.}
}

\block{Radiation Qualification Criteria}{
    \looseitems
        \item Dose and fluence targets are derived from ATLAS FLUKA/Geant4 simulations
        of the HL-LHC collision environment, scaled by layered Safety Factors (SF):
        \textbf{SF$_\text{sim}$\,=\,1.5} (simulation uncertainty),
        \textbf{SF$_\text{LDR}$\,=\,3–5} (ELDRS and lot-to-lot variation),
        \textbf{SF$_\text{beam}$\,=\,1.3} (non-monoenergetic beam spectra).
        \item Qualification targets for the Daughterboard (most-exposed region):
        \textbf{\textcolor{tid}{TID\,$\approx\,108\,\text{Gy}$}},
        \textbf{\textcolor{niel}{NIEL\,$\approx\,13.16\times10^{12}\,n_\text{eq}/\text{cm}^2$}},
        \textbf{\textcolor{see}{SEE proton fluence\,$\approx\,4.54\times10^{11}\,\text{p/cm}^2$}}.
        \item LVPS qualification targets (fingers region, highest dose zone):
        \textbf{\textcolor{tid}{TID\,$\approx\,80\,\text{Gy}$}},
        \textbf{\textcolor{niel}{NIEL\,$\approx\,8\times10^{12}\,n_\text{eq}/\text{cm}^2$}}.
    \end{itemize}
    \vspace{0.12cm}
    \begin{center}
        \includegraphics[width=1.0\linewidth]{assets/fig_radiation_criteria.pdf}
    \end{center}

    \captiontext{Safety-factor breakdown for the LVPS bricks: simulated dose
    scaled through SF$_\text{sim}$, SF$_\text{LDR}$, and SF$_\text{beam}$ to
    the final qualification target.}
}

\block{Irradiation Facilities Used}{
    \looseitems
        \item \textbf{CERN CC60} ($^{60}$Co): uniform low-dose-rate \textcolor{tid}{TID} ($\sim$3\,Gy/h), enabling ELDRS characterisation of analog circuits.
        \item \textbf{IFJ Kraków} (54\,MeV protons): combined \textcolor{tid}{TID}/\textcolor{niel}{NIEL} exposure; used for board-level DB6 campaign.
        \item \textbf{JSI TRIGA reactor}: thermal+fast neutron field for pure \textcolor{niel}{NIEL} tests ($>10^{12}$\,n$_\text{eq}$/cm$^2$/h).
        \item \textbf{PSI PIF} (10--230\,MeV protons): \textcolor{see}{SEE} cross-section and \textcolor{sel}{SEL} latch-up screening.
        \item \textbf{CERN CHARM}: mixed hadronic/EM field (p, n, $\pi$, $\gamma$) for full system-level validation.
    \end{itemize}
}

\block{The Link \& Control Daughterboard (DB6)}{
    \tightitems
        \item DB6 is the signal-processing nexus of each Minidrawer: it
        hosts 4$\times$ SFP$+$ transceivers (two 4.8\,Gbps downlinks,
        four 9.6\,Gbps uplinks), CERN \textbf{GBTx} ASICs for
        clock/command distribution, a Xilinx \textbf{Kintex UltraScale (KU)}
        FPGA running pipeline trigger and data-aggregation firmware, and a
        Microsemi \textbf{ProASIC3} radiation-hardened companion FPGA.
        \item Radiation tolerance is achieved by a \textbf{double-redundant A/B
        architecture} enforced by \textbf{TMR} state machines, \textbf{GBT-FEC}
        forward error correction, and \textcolor{sel}{SEL}-protection circuitry
        that disconnects power within $500\,\mu\text{s}$ of an over-current event.
    \end{itemize}
    \vspace{0cm}
    \begin{center}
        \includegraphics[width=1.0\linewidth]{assets/fig_db6_blockdiagram.png}
    \end{center}
    \vspace{-0.3cm}
    \captiontext{DB6 block diagram: redundant A/B halves, KU\,+\,ProASIC3 FPGAs,
    GBTx ASICs, and four SFP$+$ optical links.}
}

\block{DB6 TID, NIEL \& SEL Qualification}{
    \tightitems
        \item \textbf{\textcolor{tid}{TID}}: Boards irradiated at CERN CC60
        ($^{60}$Co, $3.37\,\text{Gy/h}$) to $220\,\text{Gy}$ and at IFJ Kraków
        (54\,MeV protons, $2.88\,\text{Gy/h}$) to $\sim\!400\,\text{Gy}$.
        Currents and temperatures remained stable throughout; no evidence of
        TID-induced degradation in digital or analog circuitry.
        \item \textbf{\textcolor{niel}{NIEL}}: DB6-0 irradiated at the TRIGA
        reactor (JSI Ljubljana) to $14\times10^{12}\,n_\text{eq}/\text{cm}^2$
        (exceeding target by $6\,\%$). KU FPGAs showed full tolerance; ProASIC3
        and FLASH memory recovered after $50^{\circ}$C anneal.
        \item \textbf{\textcolor{sel}{SEL}}: 54 and 226\,MeV proton exposures up
        to $1.11\times10^{12}\,\text{p/cm}^2$ — \textbf{zero latch-up events}
        recorded in all KU FPGAs across all boards.
    \end{itemize}
    \vspace{0cm}
    \begin{center}
        \includegraphics[width=1.0\linewidth]{assets/fig_db6_tid_test.png}
    \end{center}
    \vspace{-0.3cm}
    \captiontext{DB6-3 board currents and FPGA temperature — stable over the full
    $\sim\!400\,\text{Gy}$ proton \textcolor{tid}{TID} run (both A and B halves).}
}

\block{Component Selection \& Design Iterations}{
    \tightitems
        \item \textbf{SFP$+$ transceivers}: AVAGO modules failed at
        $<10^{12}\,n/\text{cm}^2$; 1/9 CORETEK units failed at
        $9\times10^{12}\,n/\text{cm}^2$. Selected alternatives:
        \textbf{Direktronik 33-4248} and \textbf{FS-Europe SFP-10GSR-85}.
        \item \textbf{MOSFET}: FDFMA3N109 exhibited threshold shift of
        $\Delta V_\text{th}\approx0.5\,\text{V}$ at $10^{12}\,n/\text{cm}^2$,
        disrupting the \textcolor{sel}{SEL} over-current sense. Circuit was
        removed; protection relies solely on firmware-based current limits.
        \item \textbf{Oscillators}: 4/15 LTC6902 clock oscillators failed; all
        removed and replaced with GBTx-derived clocks.
        \item All remaining components qualified. $\sim$\textbf{930 DB6 units}
        scheduled for production in \textbf{2026--2027}.
    \end{itemize}
    \vspace{0.06cm}
    \begin{center}
    \small

\block{DB6 SEU Performance \& Mitigation}{
    \looseitems
        \item FPGA configuration memories are the primary \textcolor{see}{SEU} target
        under mixed-field irradiation. Bitstream scrubbing (every $\sim$30\,ms)
        corrects upsets in-situ without interrupting the $40\,\text{MHz}$ pipeline.
        \item KU FPGA: measured SEU cross-section
        $\sigma_\text{SEU}\approx3\times10^{-14}\,\text{cm}^2$/bit → MTBF $\gg10$\,yr.
        \item ProASIC3 FLASH memory recovered fully after a $50^\circ$C anneal
        following neutron exposure to $14\times10^{12}\,n_\text{eq}/\text{cm}^2$.
        \item GBTx ASIC (custom rad-hard): zero SEU-induced errors at all test fluences.
        \item All selected SFP$+$ modules (Direktronik 33-4248,
        FS-Europe SFP-10GSR-85) demonstrated zero bit-errors up to
        $9\times10^{12}\,n/\text{cm}^2$.
    \end{itemize}
}

\block{LVPS Bricks: Design \& Optimisation}{
    \looseitems
        \item Located in the module \emph{fingers} — the highest-dose zone in
        TileCal — the LVPS steps $200\,\text{V}$ down to $10\,\text{V}$ via
        8 independent 300\,kHz two-switch forward-converter ``bricks'' per module
        ($\sim\!23\,\text{W}$ each). This segmentation contains localised failures
        without affecting adjacent channels.
        \item \textbf{Efficiency improvement}: substituting STB57N65M5 with the
        radiation-screened \textbf{SIHFS9N60A} MOSFET raised brick efficiency
        from $\sim\!58\,\%$ to $\sim\!72\,\%$, surpassing the $\geq65\,\%$
        threshold imposed by the 77\,kW cooling plant.
        \item \textbf{Wiring simplification}: a tri-state logic scheme (on/off/hold)
        halved the control lines from \textbf{18} to \textbf{9}, easing routing
        from the service cavern through the limited cable conduit.
    \end{itemize}

        \includegraphics[width=1.0\linewidth]{assets/fig_wire_reduction.pdf}

\block{LVPS Radiation Qualification}{
    \looseitems
        \item Individual LVPS components tested for \textbf{\textcolor{tid}{TID}}
        at CC60 (CERN) and PSI, for \textbf{\textcolor{niel}{NIEL}} at the TRIGA
        reactor (JSI), and for \textbf{\textcolor{see}{SEE}} at PIF (PSI proton beam).
        Fully assembled bricks then validated at \textbf{CERN CHARM} in the
        realistic mixed hadronic/EM field.
        \item Critical components — 7 rectifier diodes, \textbf{LT1681} controller,
        \textbf{LTC6241} op-amp, \textbf{LT3080} regulator, \textbf{IR2110} gate
        driver, \textbf{SIHFS9N60A} MOSFET — all passed with only minor, acceptable
        parametric shifts.
        \item \textbf{SI8920} isolation amplifier: stable for input $\leq100\,\text{mV}$;
        pre-production MOSFET lot failed at $323\,\text{Gy}$ (well above the
        $80\,\text{Gy}$ requirement, justifying final part selection);
        \textbf{HCPL7800} current-sense amplifier failed at
        $2.75\times10^{12}\,n/\text{cm}^2$ — driving replacement with SI8920.
    \end{itemize}

\block{Conclusions \& Outlook}{
    \looseitems
        \item Both the \textbf{LVPS bricks} and the \textbf{Link \& Control
        Daughterboard (DB6)} successfully satisfy all HL-LHC
        \textcolor{tid}{TID}, \textcolor{niel}{NIEL}, and \textcolor{see}{SEE}
        qualification benchmarks by comfortable margins.
        \item Radiation campaigns directly drove design optimisation: vulnerable
        components were identified and replaced (MOSFET, SFP$+$, oscillators);
        LVPS efficiency improved from $\sim\!58\,\%$ to $\sim\!72\,\%$;
        control-wire count halved.
        \item All critical components are now fully qualified. Mass production of
        $\sim\!930$ DB6 units is planned for \textbf{2026--2027}; full detector
        integration is targeted for the HL-LHC Long Shutdown 3 (LS3).
        \vspace{0.1cm}
