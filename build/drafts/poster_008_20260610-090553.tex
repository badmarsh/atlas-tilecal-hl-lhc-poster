\block{The ATLAS Tile Calorimeter \& HL-LHC Upgrade}{
    \looseitems
        \item As the primary hadronic sampling calorimeter of ATLAS, TileCal
        (built from steel absorbers and scintillating plastic tiles, covering $|\eta|<1.7$)
        is crucial for measuring hadronic jets, missing transverse energy,
        hadronically decaying $\tau$-leptons, and for muon identification.
        \item To cope with the unprecedented instantaneous luminosity and pile-up
        of the \textbf{High-Luminosity LHC}, the entire on-detector electronics system
        is being \textbf{fully replaced}.
    \end{itemize}
    \vspace{0.5cm}
    \begin{center}
    \small
    \begin{tabular}{|p{0.25\linewidth}|p{0.3\linewidth}|p{0.35\linewidth}|}
    \hline
    \textbf{Parameter} & \textbf{Current Readout} & \textbf{HL-LHC Upgrade} \\
    \hline
    \textbf{Trigger mechanism} & Analog signals to L1Calo & Fully digital at 40\,MHz \\
    \hline
    \textbf{Digitization} & On-detector, stored in pipelines & Continuous on-detector at 40\,MHz \\
    \hline
    \textbf{Data transmission} & Triggered at $\sim$100\,kHz & Untriggered continuous streaming \\
    \hline
    \textbf{Link bandwidth} & 160\,Mbps (uplink) & 4.8 / 9.6\,Gbps (downlink/uplink) \\
    \hline
    \textbf{Architecture} & 3-in-1 cards, Digitizer, Interface & Front-End Boards, Mainboard, Daughterboard \\
    \hline
    \end{tabular}
    \end{center}
    \vspace{0.2cm}
}

\block{Phase-II Minidrawer Architecture}{
    \looseitems
        \item The on-detector electronics are partitioned into \textbf{896 Minidrawers (MDs)}.
        Each MD processes signals from up to 12 PMTs at 40\,MHz. The MD houses
        Front-End Boards (signal shaping), a \textbf{Mainboard (MB)} (digitization),
        and a \textbf{Daughterboard (DB)} (control, clock distribution, and high-speed data transmission).
        Power is supplied by the \textbf{Low-Voltage Power Supply (LVPS)}.
        \item Digital streams are transmitted off-detector via multi-Gbps optical links
        to FELIX and the Tile PreProcessor (TilePPr).
    \end{itemize}
    \vspace{0.12cm}
    \begin{center}
        \includegraphics[width=1.0\linewidth]{assets/fig_minidrawer_blockdiagram.png}
    \end{center}
    \vspace{-0.2cm}
    \captiontext{TileCal Phase-II Minidrawer (left) and the on-/off-detector read-out chain (right).}
}

  \block{Single Event Upset (SEU) Rates}{
      \looseitems
          \item KU FPGAs were irradiated with 54\,MeV protons up to $2.2\times10^{11}$\,p/cm$^2$, exhibiting an SEU rate of 116\,SEU/($10^9$\,p/cm$^2$).
          \item In a worst-case scenario, this yields roughly \textbf{5498 SEUs per year}.
          \item \textbf{Mitigation}: Approximately 5487 of these are correctable by SEM and will not impact nominal runs due to TMR.
          \item The remaining 11 uncorrectable errors per year are managed by TMR and cleared by reloading the FPGA configuration from FLASH during data acquisition stops.
      \end{itemize}
      \vspace{0.12cm}
      \begin{center}
          \includegraphics[width=1.0\linewidth]{assets/fig_diodes_niel_test.pdf}
      \end{center}
      \vspace{-0.2cm}
      \captiontext{Diode forward voltage shifts under NIEL testing.}
  }

\block{Radiation Environment \& Requirements}{
    \tightitems
        \item The new hardware faces a vastly harsher radiation environment.
        Consequently, every single component must be rigorously qualified against
        \textbf{\textcolor{tid}{Total Ionizing Dose (TID)}},
        \textbf{\textcolor{niel}{Non-Ionizing Energy Loss (NIEL)}}, and
        \textbf{\textcolor{see}{Single Event Effects (SEE/SEL)}}.
        \item Radiation criteria are based on ATLAS simulations for the
        \emph{most-exposed} regions, scaled by rigorous safety factors (SF):
        \textbf{1.5} (simulation uncertainty), $\mathbf{\times}$\textbf{3--5}
        (low-dose-rate effects / lot-to-lot variation),
        and \textbf{1.3} (non-monoenergetic beams).
        \item Final Daughterboard qualification targets (with all SFs applied):
        \textbf{\textcolor{tid}{TID $\approx$ 108\,Gy}},
        \textbf{\textcolor{niel}{NIEL $\approx 13.16\times10^{12}$\,n/cm$^2$}},
        \textbf{\textcolor{see}{SEE $\approx 4.54\times10^{11}$\,h/cm$^2$}}.
    \end{itemize}
    \vspace{0.05cm}
    \begin{center}
        \includegraphics[width=1.0\linewidth]{assets/fig_radiation_criteria.pdf}
    \end{center}
    \vspace{-0.2cm}
    \captiontext{Worked example for the LVPS bricks: simulated dose/fluence scaled by
    safety factors to the qualification target.}
}

\block{The Link \& Control Daughterboard (DB6)}{
    \tightitems
        \item The DB6 achieves exceptional radiation tolerance via a
        \textbf{double-redundant} A/B architecture enforced by
        \textbf{Triple Modular Redundancy (TMR)},
        \textbf{Xilinx Soft Error Mitigation (SEM)},
        \textbf{GBT-FEC} error correction, and \textcolor{sel}{SEL}-protection circuitry.
        \item Core components include 4$\times$ commercial \textbf{SFP+} transceivers,
        custom CERN \textbf{GBTx} ASICs, Xilinx \textbf{Kintex UltraScale (KU)}
        and Microsemi \textbf{ProASIC3} FPGAs.
    \end{itemize}
    \vspace{0cm}
    \begin{center}
        \includegraphics[width=1.0\linewidth]{assets/fig_db6_blockdiagram.png}
    \end{center}
    \vspace{-0.3cm}
    \captiontext{DB6 block diagram: two redundant halves, KU + ProASIC FPGAs, GBTx and four SFP$+$ optical links.}
}

  \block{MOSFET Shifts \& SEL Mitigation}{
      \looseitems
          \item The originally selected FDFMA3N109 MOSFETs displayed notable threshold voltage shifts. These shifts were directly correlated to the \textbf{195\,Gy} Gamma component measured in the reactor cavity.
          \item Consequently, the supplementary over-current protection circuitry---which relied heavily on these MOSFETs for \textcolor{sel}{SEL} mitigation---had to be entirely removed from the final DB6 design iteration.
      \end{itemize}
      \vspace{0.12cm}
      \begin{center}
          \includegraphics[width=1.0\linewidth]{assets/fig_mosfet_threshold.png}
      \end{center}
      \vspace{-0.2cm}
      \captiontext{Threshold voltage shifts depicted in IV curves of irradiated FDFMA3N109 MOSFETs.}
  }

\block{Daughterboard Qualification}{
    \tightitems
        \item \textbf{\textcolor{tid}{TID}}: Boards irradiated at CERN CC60 ($^{60}$Co)
        to 220\,Gy and at IFJ Kraków to $\sim$400\,Gy. Temperatures and currents stable throughout.
        \item \textbf{\textcolor{niel}{NIEL}}: DB6-0 irradiated at TRIGA reactor
        to $14\times10^{12}$\,n$_\text{eq}$/cm$^2$. KU FPGAs showed
        strong tolerance; ProASIC and FLASH recovered after annealing.
        \item \textbf{\textcolor{sel}{SEL}}: Tests with 54\,MeV and 226\,MeV protons
        yielded \textbf{zero single-event latch-ups} in all KU FPGAs.
    \end{itemize}
    \vspace{0cm}
    \begin{center}
        \includegraphics[width=1.0\linewidth]{assets/fig_db6_tid_test.png}
    \end{center}
    \vspace{-0.3cm}
    \captiontext{DB6-3 board currents and FPGA temperature, stable over the full $\sim$400\,Gy proton \textcolor{tid}{TID} run.}
}

\block{Low-Voltage Power Supply (LVPS) Design}{
    \looseitems
        \item Located in the module \emph{fingers}, the LVPS experiences the most severe
        radiation environment within TileCal. The Phase-II design deploys \textbf{8 independent
        DC-DC ``bricks''} per module: each a 300\,kHz two-switch forward converter stepping
        \textbf{200\,V down to 10\,V} ($\sim$23\,W).
        \item \textbf{Enhanced Efficiency}: Replacing the STB57N65M5 MOSFET with
        \textbf{SIHFS9N60A} improved brick efficiency from $\sim$58\% to \textbf{$\sim$72\%},
        surpassing the $\geq$65\% threshold.
        \item \textbf{Simplified Wiring}: Control signals halved from \textbf{18} to \textbf{9}
        via a tri-state logic scheme (on/off/hold).
    \end{itemize}
    \vspace{0.1cm}
    \begin{center}
        \includegraphics[width=1.0\linewidth]{assets/fig_lvps_brick.pdf}\\[2mm]
        \includegraphics[width=1.0\linewidth]{assets/fig_wire_reduction.pdf}
    \end{center}
    \vspace{-0.3cm}
    \captiontext{LVPS brick diagram and top view (top); 18-to-9 control-wire reduction (bottom).}
}

  \block{LVPS Testing Campaign}{
      \looseitems
          \item Critical components (LT1681, LTC6241, IR2110) passed TID/NIEL tests with minor shifts. System-level validation occurred at CERN CHARM.
      \end{itemize}
  }

\block{LVPS Radiation Qualification}{
    \looseitems
        \item Components tested for \textbf{\textcolor{tid}{TID}} (CC60/PSI),
        \textbf{\textcolor{niel}{NIEL}} (TRIGA), and \textbf{\textcolor{see}{SEE}} (PIF);
        assembled bricks validated in \textbf{system tests at CERN CHARM}.
        \item Critical parts (7 diodes, \textbf{LT1681}, \textbf{LTC6241},
        \textbf{LT3080}, \textbf{IR2110}, \textbf{SIHFS9N60A}) all passed with only minor shifts.
        \item \textbf{SI8920} isolation amplifier is stable when input $\leq$100\,mV;
        pre-production MOSFETs failed at 323\,Gy (well beyond the 80\,Gy req.);
        HCPL7800 failed at $2.75\times10^{12}$\,n/cm$^2$ — all dictated final selections.
    \end{itemize}

\block{Conclusions \& Outlook}{
    \looseitems
        \item The \textbf{Phase-II Readout System} for ATLAS TileCal introduces a fully digital
        readout architecture capable of 40\,MHz continuous streaming to the off-detector processors.
        \item Both \textbf{LVPS bricks} and the \textbf{Link Daughterboard}
        successfully satisfy HL-LHC \textcolor{tid}{TID}, \textcolor{niel}{NIEL}, and
        \textcolor{see}{SEE} benchmarks.
        \item \textbf{Project Outlook}: Mass production of $\sim$930 Daughterboards
        is slated for 2026--2027. \item System-level testing validated both modules independently.
    \end{itemize}
}

  \block{Conclusions \& Mass Production}{
      \looseitems
          \item The extensive and multi-layered testing campaign confirmed full HL-LHC compatibility for the vast majority of DB6 and LVPS electronic components.
          \item Radiation testing results successfully informed critical and targeted design changes---such as the removal of problematic oscillators and MOSFETs---achieving optimal systemic reliability.
          \item With qualifications nearly complete, approximately \textbf{930 DB6 units} will enter full-scale mass production starting in 2026.
      \end{itemize}
  }
